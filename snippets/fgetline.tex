\begin{DoxyDocu}{fgetline}
\label{numberline_8c_a8b39e07a1bd3338f27e7dacabeafde15_a8b39e07a1bd3338f27e7dacabeafde15}
read one text line of at most \code{size-1} bytes.

The \code{'\\n'} character is replaced by {\ttfamily 0}.

\begin{DoxyReturn}{Returns}
s if an entire line was read successfully. Otherwise, \code{0} is returned and {\itshape s} contains a maximal partial line that could be read. {\itshape s} is null terminated.
\end{DoxyReturn}


\end{DoxyDocu}
